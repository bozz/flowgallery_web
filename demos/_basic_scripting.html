var $gallery = $('#gallery').flowgallery({
  easing: 'easeOutCubic'
});

var flowgallery = $gallery.data('flowgallery');

$('#d4-prev-button').click(function() {
  flowgallery.prev();
});

$('#d4-next-button').click(function() {
  flowgallery.next();
});

$('#d4-goto-select').change(function() {
  flowgallery.goto( $(this).val() );
});

$('#d4-enabled-cb').click(function() {
  if(gallery.isEnabled()) {
    flowgallery.disable();
  } else {
    flowgallery.enable();
  }
});
